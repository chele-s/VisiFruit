# ========== NOTAS DE INSTALACIÓN ==========
# 
# Para Raspberry Pi 5:
# -------------------
# 1. Actualiza el sistema primero:
#    sudo apt-get update && sudo apt-get upgrade
#
# 2. Instala dependencias del sistema:
#    sudo apt-get install python3-pip python3-dev libatlas-base-dev
#    sudo apt-get install libopencv-dev python3-opencv
#
# 3. Instala PyTorch optimizado para Raspberry Pi (CPU):
#    pip3 install torch torchvision --index-url https://download.pytorch.org/whl/cpu
#
# 4. Instala el resto de dependencias:
#    pip3 install -r requirements_yolov8.txt
#
# 5. Verifica la instalación:
#    python3 -c "from ultralytics import YOLO; print('YOLOv8 instalado correctamente')"
#

# ---------------------------------
# pip install -r requirements_yolov8.txt
